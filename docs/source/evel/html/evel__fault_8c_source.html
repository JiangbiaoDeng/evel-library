418 void evel_free_fault(EVENT_FAULT * event)
419 {
420  FAULT_ADDL_INFO * addl_info = NULL;
421 
422  EVEL_ENTER();
423 
424  /***************************************************************************/
425  /* Check preconditions. As an internal API we don't allow freeing NULL */
426  /* events as we do on the public API. */
427  /***************************************************************************/
428  assert(event != NULL);
429  assert(event->header.event_domain == EVEL_DOMAIN_FAULT);
430 
431  /***************************************************************************/
432  /* Free all internal strings then the header itself. */
433  /***************************************************************************/
434  addl_info = dlist_pop_last(&event->additional_info);
435  while (addl_info != NULL)
436  {
437  EVEL_DEBUG("Freeing Additional Info (%s, %s)",
438  addl_info->name,
439  addl_info->value);
440  free(addl_info->name);
441  free(addl_info->value);
442  free(addl_info);
443  addl_info = dlist_pop_last(&event->additional_info);
444  }
445  free(event->alarm_condition);
446  free(event->alarm_interface_a);
447  free(event->specific_problem);
448  evel_free_header(&event->header);
449 
450  EVEL_EXIT();
451 }
