395 {

416  free(feature_use->feature_id);

419  }
420  measurement_group = dlist_pop_last(&event->measurement_groups);
421  while (measurement_group != NULL)
422  {
423  EVEL_DEBUG("Freeing Measurement Group (%s)", measurement_group->name);
424 
425  custom_measurement = dlist_pop_last(&measurement_group->measurements);
426  while (custom_measurement != NULL)
427  {
428  EVEL_DEBUG("Freeing mesaurement (%s)", custom_measurement->name);
429 
430  free(custom_measurement->name);
