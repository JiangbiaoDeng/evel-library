493 static EVEL_ERR_CODES evel_post_api(char * msg, size_t size)
494 {
495  int rc = EVEL_SUCCESS;
496  CURLcode curl_rc = CURLE_OK;
497  MEMORY_CHUNK rx_chunk;
498  MEMORY_CHUNK tx_chunk;
499  int http_response_code = 0;
500 
501  EVEL_ENTER();
502 
503  /***************************************************************************/
504  /* Create the memory chunk to be used for the response to the post. The */
505  /* will be realloced. */
506  /***************************************************************************/
507  rx_chunk.memory = malloc(1);
508  assert(rx_chunk.memory != NULL);
509  rx_chunk.size = 0;
510 
511  /***************************************************************************/
512  /* Create the memory chunk to be sent as the body of the post. */
513  /***************************************************************************/
514  tx_chunk.memory = msg;
515  tx_chunk.size = size;
516  EVEL_DEBUG("Sending chunk of size %d", tx_chunk.size);
517 
518  /***************************************************************************/
519  /* Point to the data to be received. */
520  /***************************************************************************/
521  curl_rc = curl_easy_setopt(curl_handle, CURLOPT_WRITEDATA, (void *)&rx_chunk);
522  if (curl_rc != CURLE_OK)
523  {
524  rc = EVEL_CURL_LIBRARY_FAIL;
525  log_error_state("Failed to initialize libCURL to upload. "
526  "Error code=%d (%s)", curl_rc, curl_err_string);
527  goto exit_label;
528  }
529  EVEL_DEBUG("Initialized data to receive");
530 
531  /***************************************************************************/
532  /* pointer to pass to our read function */
533  /***************************************************************************/
534  curl_rc = curl_easy_setopt(curl_handle, CURLOPT_READDATA, &tx_chunk);
535  if (curl_rc != CURLE_OK)
536  {
537  rc = EVEL_CURL_LIBRARY_FAIL;
538  log_error_state("Failed to set upload data for libCURL to upload. "
539  "Error code=%d (%s)", curl_rc, curl_err_string);
540  goto exit_label;
541  }
542  EVEL_DEBUG("Initialized data to send");
543 
544  /***************************************************************************/
545  /* Size of the data to transmit. */
546  /***************************************************************************/
547  curl_rc = curl_easy_setopt(curl_handle,
548  CURLOPT_POSTFIELDSIZE,
549  tx_chunk.size);
550  if (curl_rc != CURLE_OK)
551  {
552  rc = EVEL_CURL_LIBRARY_FAIL;
553  log_error_state("Failed to set length of upload data for libCURL to "
554  "upload. Error code=%d (%s)", curl_rc, curl_err_string);
555  goto exit_label;
556  }
557  EVEL_DEBUG("Initialized length of data to send");
558 
559  /***************************************************************************/
560  /* Now run off and do what you've been told! */
561  /***************************************************************************/
562  curl_rc = curl_easy_perform(curl_handle);
563  if (curl_rc != CURLE_OK)
564  {
565  rc = EVEL_CURL_LIBRARY_FAIL;
566  log_error_state("Failed to transfer an event to Vendor Event Listener! "
567  "Error code=%d (%s)", curl_rc, curl_err_string);
568  EVEL_ERROR("Dropped event: %s", msg);
569  goto exit_label;
570  }
571 
572  /***************************************************************************/
573  /* See what response we got - any 2XX response is good. */
574  /***************************************************************************/
575  curl_easy_getinfo(curl_handle, CURLINFO_RESPONSE_CODE, &http_response_code);
576  EVEL_DEBUG("HTTP response code: %d", http_response_code);
577  if ((http_response_code / 100) == 2)
578  {
579  /*************************************************************************/
580  /* If the server responded with data it may be interesting but not a */
581  /* problem. */
582  /*************************************************************************/
583  if (rx_chunk.size > 0)
584  {
585  EVEL_DEBUG("Server returned data unexpectedly = %d (%s)",
586  rx_chunk.size,
587  rx_chunk.memory);
588  }
589  }
590  else
591  {
592  EVEL_ERROR("Unexpected HTTP response code: %d with data size %d (%s)",
593  http_response_code,
594  rx_chunk.size,
595  rx_chunk.size > 0 ? rx_chunk.memory : "NONE");
596  EVEL_ERROR("Potentially dropped event: %s", msg);
597  }
598 
599 exit_label:
600  free(rx_chunk.memory);
601  EVEL_EXIT();
602  return(rc);
603 }

614 {

648 size_t evel_write_callback(void *contents,
649  size_t size,
650  size_t nmemb,
651  void *userp)
652 {
653  size_t realsize = size * nmemb;
654  MEMORY_CHUNK *rx_chunk = (MEMORY_CHUNK *)userp;
655 
656  EVEL_ENTER();
657 
658  EVEL_DEBUG("Called with %d chunks of %d size = %d", nmemb, size, realsize);
659  EVEL_DEBUG("rx chunk size is %d", rx_chunk->size);
660 
661  rx_chunk->memory = realloc(rx_chunk->memory, rx_chunk->size + realsize + 1);
662  if(rx_chunk->memory == NULL) {
663  /* out of memory! */
664  printf("not enough memory (realloc returned NULL)\n");
665  return 0;
666  }
667 
668  memcpy(&(rx_chunk->memory[rx_chunk->size]), contents, realsize);
669  rx_chunk->size += realsize;
670  rx_chunk->memory[rx_chunk->size] = 0;
671 
672  EVEL_DEBUG("Rx data: %s", rx_chunk->memory);
673  EVEL_DEBUG("Returning: %d", realsize);
674 
675  EVEL_EXIT();

679 /**************************************************************************/
687 static void * event_handler(void * arg __attribute__ ((unused)))
688 {
689  int old_type = 0;
690  EVENT_HEADER * msg = NULL;
691  EVENT_INTERNAL * internal_msg = NULL;
692  int json_size = 0;
693  char json_body[EVEL_MAX_JSON_BODY];
694  int rc = EVEL_SUCCESS;
695 
696 
697  EVEL_INFO("Event handler thread started");
698 
699  /***************************************************************************/
700  /* Set this thread to be cancellable immediately. */
701  /***************************************************************************/
702  pthread_setcanceltype(PTHREAD_CANCEL_ASYNCHRONOUS, &old_type);
703 
704  /***************************************************************************/
705  /* Set the handler as active, defending against weird situations like */
706  /* immediately shutting down after initializing the library so the */
707  /* handler never gets started up properly. */
708  /***************************************************************************/
709  if (evt_handler_state == EVT_HANDLER_INACTIVE)
710  {
711  evt_handler_state = EVT_HANDLER_ACTIVE;
712  }
713  else
714  {
715  EVEL_ERROR("Event Handler State was not INACTIVE at start-up - "
716  "Handler will exit immediately!");
717  }
718 
719  while (evt_handler_state == EVT_HANDLER_ACTIVE)
720  {
721  /*************************************************************************/
722  /* Wait for a message to be received. */
723  /*************************************************************************/
724  EVEL_DEBUG("Event handler getting any messages");
725  msg = ring_buffer_read(&event_buffer);
726 
727  /*************************************************************************/
728  /* Internal events get special treatment while regular events get posted */
729  /* to the far side. */
730  /*************************************************************************/
731  if (msg->event_domain != EVEL_DOMAIN_INTERNAL)
732  {
733  EVEL_DEBUG("External event received");
734 
735  /*************************************************************************/
736  /* Encode the event in JSON. */
737  /*************************************************************************/
738  json_size = evel_json_encode_event(json_body, EVEL_MAX_JSON_BODY, msg);
739 
740  /*************************************************************************/
741  /* Send the JSON across the API. */
742  /*************************************************************************/
743  EVEL_DEBUG("Sending JSON of size %d is: %s", json_size, json_body);
744  rc = evel_post_api(json_body, json_size);
745  if (rc != EVEL_SUCCESS)
746  {
747  EVEL_ERROR("Failed to transfer the data. Error code=%d", rc);
748  }
749  }
750  else
751  {
752  EVEL_DEBUG("Internal event received");
753  internal_msg = (EVENT_INTERNAL *) msg;
754  assert(internal_msg->command == EVT_CMD_TERMINATE);
755  evt_handler_state = EVT_HANDLER_TERMINATING;
756  }
757 
758  /*************************************************************************/
759  /* We are responsible for freeing the memory. */
760  /*************************************************************************/
761  evel_free_event(msg);
762  msg = NULL;
763  }
764 
765  /***************************************************************************/
766  /* The event handler is now exiting. The ring-buffer could contain events */
767  /* which have not been processed, so deplete those. Because we've been */
768  /* asked to exit we can be confident that the foreground will have stopped */
769  /* sending events in so we know that this process will conclude! */
770  /***************************************************************************/
771  evt_handler_state = EVT_HANDLER_TERMINATING;
772  while (!ring_buffer_is_empty(&event_buffer))
773  {
774  EVEL_DEBUG("Reading event from buffer");
775  msg = ring_buffer_read(&event_buffer);
776  evel_free_event(msg);
777  }
778  evt_handler_state = EVT_HANDLER_TERMINATED;
779  EVEL_INFO("Event handler thread stopped");
780 
781  return (NULL);
782 }
